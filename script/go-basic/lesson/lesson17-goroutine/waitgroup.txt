func task(id int, wg *sync.WaitGroup) {
	defer wg.Done()
	fmt.Printf("Task %d bat dau \n", id)
	time.Sleep(1 * time.Second)
	fmt.Printf("Task %d ket thuc \n", id)
}

func main() {
	start := time.Now()
	var wg sync.WaitGroup

	for i := 1; i <= 4; i++ {
		wg.Add(1)
		go task(i, &wg)
	}

	wg.Wait()

	fmt.Println("Thoi gian thuc hien: ", time.Since(start))
}
