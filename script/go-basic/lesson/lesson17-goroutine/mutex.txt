package main

import (
	"fmt"
	"sync"
)

func main() {
	var wg sync.WaitGroup
	var mu sync.Mutex
	tokens := 0

	for i := 1; i <= 1000; i++ {
		wg.Add(1)
		go func() {
			mu.Lock()
			tokens++
			mu.Unlock()

			wg.Done()
		}()
	}

	wg.Wait()

	fmt.Println(tokens) // 1000
}
