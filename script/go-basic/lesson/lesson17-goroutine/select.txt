package main

import (
	"fmt"
	"time"
)

func main() {
	ch1 := make(chan string)
	ch2 := make(chan string)

	go func() {
		time.Sleep(5 * time.Second)
		ch1 <- "Data from channel 1"
	}()

	go func() {
		time.Sleep(1 * time.Second)
		ch2 <- "Data from channel 2"
	}()

	for i := 1; i <= 2; i++ {
		select {
		case msg1 := <-ch1:
			fmt.Println("msg1", msg1)
		case msg2 := <-ch2:
			fmt.Println("msg2", msg2)
		}
	}
}
